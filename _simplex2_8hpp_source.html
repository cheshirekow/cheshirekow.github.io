110 template <typename Output1, typename Output2, typename Output3>
111 inline
112 void SimplexOps<Traits>::vsetSplit( Simplex& Sa, Simplex& Sb,
113  Output1 a_only, Output2 b_only, Output3 common)
114 {
115  PointRef* first1 = Sa.V;
116  PointRef* last1 = first1 + NDim+1;
117  PointRef* first2 = Sb.V;
118  PointRef* last2 = first2 + NDim+1;
119 
120  while( first1 != last1 && first2 != last2 )
121  {
122  if(*first1 < *first2)
123  *a_only = *first1++;
124  else if( *first2 < *first1 )
125  *b_only = *first2++;
126  else
127  {
128  *common++ = *first1++;
129  ++first2;
130  }
131  }
132  std::copy(first1,last1,a_only);
133  std::copy(first2,last2,b_only);
134 }

209 template <class Traits>
210 inline
211 void SimplexOps<Traits>::finish( Simplex& S )
212 {
213  PointRef peak = S.V[S.iPeak];
214  std::map<PointRef,SimplexRef> kv;
215 
216 
217  for( int i=0; i < NDim+1; i++ )
218  kv[ S.V[i] ] = S.N[i];
219 
220  int i=0;
221  for( auto pair: kv )
222  {
223  if( pair.first == peak )
224  S.iPeak = i;
225  S.V[i] = pair.first;
226  S.N[i] = pair.second;
227  i++;
228  }
229 }

231 template <class Traits>
232 inline
233 void SimplexOps<Traits>::computeBase( Simplex& S, Deref& deref )
234 {
235  typedef Eigen::Matrix<Scalar,NDim,NDim> Matrix;
236  typedef Eigen::Matrix<Scalar,NDim,1> Vector;
237 
238  Matrix A;
239  Vector b;
240 
241  unsigned int j=0;
242  for(unsigned int i=0; i < NDim+1; i++)
243  if( i != S.iPeak )
244  A.row(j++) = deref.point(S.V[i]);
245  b.setConstant(1);
246 
247  // solve for the normal
248  S.n = A.fullPivLu().solve(b);
249  S.n.normalize();
250 
251  // and then find the value of 'c' (hyperplane offset)
252  j = S.iPeak == 0 ? 1 : 0;
253  S.o = deref.point(S.V[j]).dot(S.n);
254 }

256 template <class Traits>
257 inline
258 void SimplexOps<Traits>::
259  orientBase( Simplex& S, const Point& x, Orientation orient )
260 {
261  // then orient the hyperplane so that the peak vertex is on the
262  // "positive" side (i.e. the 'less than' side)
263 
264  switch( orient )
265  {
266  case INSIDE:
267  {
268  if( S.n.dot( x ) > S.o )
269  {
270  S.n = -S.n;
271  S.o = -S.o;
272  }
273  break;
274  }
275 
276  case OUTSIDE:
277  {
278  if( S.n.dot( x ) < S.o )
279  {
280  S.n = -S.n;
281  S.o = -S.o;
282  }
283  break;
284  }
285 
286  default:
287  assert(false);
288  break;
289  }
290 }
