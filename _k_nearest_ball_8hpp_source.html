95 template <class Traits,template<class> class Allocator>
96 bool KNearestBall<Traits,Allocator>::shouldRecurse(
97  const Point_t& q, const HyperRect_t& r )
98 {
99  // check if we have to recurse into the farther subtree by
100  // finding out if the nearest point in the hyperrectangle is
101  // closer to the query point then the current k'th best guess
102  // or if there are less then k nearest in the queue
103  Format_t dist2 = m_dist2(q,r);
104  bool isInRange = dist2 < m_radius*m_radius;
105  bool isCloser = dist2 < m_queue.rbegin()->d2;
106  bool isAvailability = m_queue.size() < m_k;
107 
108  return ( isInRange && ( isCloser || isAvailability) );
109 }
