63  {
64  max_coeff_numerator = intlist::max< SpecNum >::value,
65  max_coeff_denominator = intlist::max< SpecDen >::value,
66  max_coeff_quotient = max_coeff_numerator - max_coeff_denominator,
67  max_coeff_remainder = max_coeff_numerator,
68  size_numerator = max_coeff_numerator + 1,
69  size_denominator = max_coeff_denominator + 1
70  };

78  template< int size, int min >
79  struct QuotientWork
80  {
81  __host__ __device__
82  static void work( Polynomial<Scalar,SpecQuot>& q,
83  Polynomial<Scalar,SpecRem>& r,
84  ExpNum const& N, ExpDen const& D )
85  {
86  // degree of the new term
87  enum{ n = size - size_denominator };
88 
89  // coefficient of the new term
90  Scalar t = get<size-1>(r) /
91  get<max_coeff_denominator>(D);
92 
93  // append the new term to the quotient
94  set<n>(q) = t;
95 
96  // update the remainder
97  r = N - q*D;
98  }
99  };
