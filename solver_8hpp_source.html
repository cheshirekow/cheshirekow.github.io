34 struct SolverIterator {
35  typedef Path<Format_t> Result_t;

39  static Result_t solve(const Eigen::Matrix<Format_t, 3, 1>& q0,
40  const Eigen::Matrix<Format_t, 3, 1>& q1,
41  const Format_t r) {
