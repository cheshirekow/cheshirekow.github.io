58 struct SolverIterator {

60  typedef HyperRect<Format_t> Hyper_t;
61  typedef Path<Format_t> Result_t;
62  typedef Integrate<Format_t> Integrate_t;
63 
64  typedef SolverIterator<xSpec, ySpec, tSpec + 1, Format_t> Next_t;
65  typedef Solver<xSpec, ySpec, tSpec, Format_t> Solver_t;
66  typedef SpecPack<xSpec, ySpec, tSpec> Spec;
67 
68  static Result_t solve(const Vector3d_t& q0, const Hyper_t& h,
69  const Format_t r) {
70  Result_t thisSoln = Solver_t::solve(q0, h, r);
71 
72  // if the solution is nominally feasible, then check to see if it
73  // satisfies all non-explicit constraints
74  if (thisSoln.f) {
75  Vector3d_t q1 = Integrate_t::solve(q0, thisSoln, r);
76  thisSoln.f = validate<Format_t, xSpec, ySpec, tSpec>(q1, h);
77  }
78 
79  Result_t thisAccum = bestOf(thisSoln, Next_t::solve(q0, h, r), r);
80 
81  // std::cout << "solver iterator, spec " << Spec::Result << ", best:
82  // "
83  // << thisSoln.dist(r) << " " << (thisSoln.f ? "T":"F")
84  // << "\n";
85 
86  return thisAccum;
87  };
88 };
