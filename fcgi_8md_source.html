1 C++ Bindings for fact-cgi {#cpp_fcgi_page}
2 ==========
3 
4 cpp-fcgi provides a thin object oriented wrapper around the thread-safe
5 fast CGI library API. The library is rather object oriented to begin with,
6 so for the most part it just provides member methods which wrap the
7 object methods from the C library. There isn't any added functionality,
8 it just makes code slightly more readable.
9 
10 Below is a simple example of it's usage:
11 
12 @code{cc}
13 
14 #include <cpp-fcgi.h>
15 #include <iostream>
16 
17 int main( int argc, char** argv )
18 {
19  if( argc < 2 )
20  {
21  std::cerr << "Usage: " << argv[0] << " [SOCKETPATH]|:[PORTNUMBER] \n";
22  return 0;
23  }
24 
25  fcgi::Socket socket;
26  fcgi::Request request;
27 
28  // initialize the library (same as FCGX_Init() )
29  int result = fcgi::init();
30  if( result != 0 )
31  {
32  std::cerr << "Failed to initialize FCGI" << std::endl;
33  return 0;
34  }
35 
36  // open the socket (same as FCGX_OpenSocket )
37  result = socket.open(argv[1],100);
38  if( result < 0 )
39  {
40  std::cerr << "Failed to open socket: " << argv[1] << ", error: "
41  << result << "\n";
42  return 0;
43  }
44 
45  // initialize the request object and associate it with the opened socket
46  if( request.init(socket) != 0 )
47  {
48  socket.close();
49  std::cerr << "Failed to initialize the request\n";
50  return 0;
51  }
52 
53  // just a count of requests processed
54  int reqNum = 0;
55 
56  // loop and receive requests
57  while( true )
58  {
59  // accept a request
60  result = request.accept();
61 
62  // check the result
63  if( result != 0 )
64  {
65  std::cerr << "Failed to accept request: " << result << std::endl;
66  break;
67  }
68 
69  // write a simple message
70  request.out() << "Content-type: text/html\n"
71  << "\n"
72  << "<title>FastCGI Hello! (C, fcgiapp library)</title>"
73  << "<h1>FastCGI Hello! (C, fcgiapp library)</h1>"
74  << "Request number " << ++reqNum
75  << " running on host <i>"
76  << request.getParam("SERVER_NAME")
77  << "</i> Process ID: "
78  << getpid()
79  << "\n";
80 
81  // free request memory and prepare for the next one, note this is
82  // not stricly necessary as it will happen automatically on the next
83  // accept() call if the request hasn't been finished yet
84  request.finish();
85  }
86 
87  request.free(); //< free hidden memory of the request object
88  socket.close(); //< close the socket
89 }
90 
91 @endcode
