98 
99  int count0 = 0;
100  int count1 = 1;
101

103  typename types::Poly0 p0 = normalized(poly);
104  val0a = polyval(p0,s0);
105  val1a = polyval(p0,s1);
106 
107  typename types::Poly1 p1 = normalized( d_dx(poly) );
108  val0b = polyval(p1,s0);
109  val1b = polyval(p1,s1);
110  if( sturm_detail::sgn(val0a) != sturm_detail::sgn(val0b) )
111  count0++;
112  if( sturm_detail::sgn(val1a) != sturm_detail::sgn(val1b) )
113  count1++;
114  val0a = val0b;
115  val1a = val1b;
116 
117  return abs(count1-count0);
118  }
