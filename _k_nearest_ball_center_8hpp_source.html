82 template <class Traits, template<class> class Allocator>
83 void KNearestBallCenter<Traits,Allocator>::evaluate(
84  const Point_t& q, const Point_t& p, Node_t* n)
85 {
86  Format_t dist2c = m_dist2(m_center,p);
87  Format_t r2 = m_radius*m_radius;
88 
89  // if this point is within the search ball
90  if( dist2c < r2 )
91  {
92  Format_t dist2 = m_dist2(q,p);
93 
94  Key_t key;
95  key.d2 = dist2;
96  key.n = n;
97 
98  m_queue.insert(key);
99 
100  if( m_queue.size() > m_k )
101  m_queue.erase( --m_queue.end() );
102  }
103 }

107 template <class Traits, template<class> class Allocator>
108 bool KNearestBallCenter<Traits,Allocator>::shouldRecurse(
109  const Point_t& q, const HyperRect_t& r )
110 {
111  // check if we have to recurse into the farther subtree by
112  // finding out if the nearest point in the hyperrectangle is
113  // closer to the query point then the current k'th best guess
114  // or if there are less then k nearest in the queue
115  Format_t dist2c = m_dist2(m_center,r);
116  Format_t dist2 = m_dist2(q,r);
117  Format_t r2 = m_radius*m_radius;
118  bool isInRange = dist2c < r2;
119  bool isCloser = m_queue.size() > 0 &&
120  ( dist2 < m_queue.rbegin()->d2 );
121  bool isAvailability = m_queue.size() < m_k;
122 
123  return ( isInRange && ( isCloser || isAvailability) );
124 }
