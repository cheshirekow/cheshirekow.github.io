34 inline Scalar Norm(_RValue<Scalar, Exp> const& A) {
35  Scalar r(0);
36  for (int i = 0; i < A.size(); i++) {
37  r += A[i] * A[i];
38  }
39  return std::sqrt(r);
40 }
