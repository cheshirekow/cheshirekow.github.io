142 template <typename KeyType, typename ValueType>
143 uint_t Sorter<KeyType,ValueType>::sort(
144  KeyType *d_DstKey,
145  ValueType *d_DstVal,
146  KeyType *d_SrcKey,
147  ValueType *d_SrcVal,
148  uint_t arrayLength,
149  Direction dir
150  )
151 {
152  // if the array is not a power of two, then find the next higher power
153  // of two for the array length and prepare everything from the end of
154  // the array up to that power of two length (by prepare, I mean set them
155  // all to the max or min value, i.e. zero or inf for unsigned integers).
156  uint_t actualArrayLength = arrayLength;
157  arrayLength = nextPow2(arrayLength);
158  uint_t prepareThreads = arrayLength - actualArrayLength;
159 
160  if(prepareThreads)
161  this->prepare(d_SrcKey+actualArrayLength, prepareThreads, dir);
162 
163  uint_t threadCount =
164  bitonic::sort<KeyType,ValueType>(
165  d_DstKey, d_DstVal,
166  d_SrcKey, d_SrcVal,
167  arrayLength, m_sharedLength, dir,
168  m_threadsMerge);
169 
170  // will throw an exception if the kernel call failed
171  cuda::checkLastError("bitonic::sort");
172 
173  return threadCount;
174 }

182  Direction dir)
183 {

190  }

201  m_max,
202  arrayLength);
203  }
204  else
205  {
206 // std::cout << "Preparing sort by padding with: " << m_min << "\n";
207  bitonic::prepare<KeyType><<<blockCount,threadCount>>>(d_SrcKey,
208  m_min,
